    # import
    import hypertools as hyp
    import numpy as np

    # load example data
    geo = hyp.load('weights_sample')
    data = geo.get_data()

    # simulate labels
    labels=[]
    for idx,i in enumerate(data):
        tmp=[]
        for iidx,ii in enumerate(i):
            if iidx==0:
                tmp.append('Subject ' + str(idx))
            else:
                tmp.append(None)
        labels.append(tmp)

    # plot
    geo.plot(fmt='.', labels=labels)
