    from sklearn import datasets
    import hypertools as hyp

    digits = datasets.load_digits(n_class=6)
    data = digits.data
    hue = digits.target.astype('str')

    hyp.plot(data, '.', reduce='UMAP', hue=hue, ndims=2)
